#include <iostream>
#include <time.h>
#include <cstdlib>

using namespace std;


int main()
{
    setlocale(LC_ALL, "rus");
    srand(time(NULL));
    int n;//количество элементов в массиве
    cin >> n;
    int k = n;
    int arrays[n];
    int counter[n];

    //считываем массив
    for(int i = 0; i < n; i++)
        arrays[i]= 1+rand()%10;

     // отображение не отсортированного массива
     for(int i = 0; i < n; i++)
            cout << arrays[i]<<". ";

    // обнуляем массив счетчик
    for(int i = 0; i < n; i++)
        counter[i] =0;

    //строим массив с
    for(int i = 0; i < n; i++)
    {
        counter[arrays[i]]++;
    }

    //бежимся по всему отрезку
    //с 0 до k-1
   cout << endl;
    for(int i = 0; i < k; i++)
    {
        //выводим i c[i] раз
        for(int j = 0; j < counter[i]; j++)
            cout << i << ", ";
    }
    return 0;
}
